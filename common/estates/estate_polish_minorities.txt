estate_polish_minorities = {
	icon = 3
	
	color = { 85 155 132 }
	
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no
	
	# Estate will always be removed when conquered by a different religion than previous owner
	always_remove_on_religious_conquest = yes
	
	# If true, country will get estate
	trigger = {
		has_country_flag = allow_polish_estate_ME
	}
	
	province_trigger = {
		culture = polish
	}
	
	start_territory_weight = 25
	
	min_autonomy = 30
	
	autonomy_does_not_affect = local_manpower_modifier
	
	country_modifier_happy = {
		production_efficiency = 0.1
		manpower_recovery_speed = 0.15
		cavalry_cost = -0.1
	}
	country_modifier_neutral = {
		production_efficiency = 0.05
		manpower_recovery_speed = 0.1
	}
	country_modifier_angry = {
		cavalry_cost = 0.05
		manpower_recovery_speed = -0.1
		production_efficiency = -0.05
	}
	
	
	province_modifier_happy = {
		local_unrest = -10
		local_build_cost = -0.15
	}
	province_modifier_neutral = {
		local_unrest = -5
	}	
	province_modifier_angry = {
		local_unrest = 8
		local_tax_modifier = -0.1
	}	
	
	base_influence = 20
	
	influence_modifier = {
		desc = EST_VAL_ACCEPTED_CULTURE
		trigger = {
			accepted_culture = polish
		}
		influence = 10
	}

		loyalty_modifier = {
			desc = EST_VAL_ACCEPTED
			trigger = {
				accepted_culture = polish
			}
			loyalty = 5
		}
		
		loyalty_modifier = {
			desc = EST_VAL_not_ACCEPTED
			trigger = {
				accepted_culture = polish
			}
			loyalty = -5
		}
		
	
	interaction = {
		name = "POLISH_NATIONALISM"
		icon = 38
		
		years_between_use = 20
		
		potential = {
			always = yes
		}
		
		trigger = {
			accepted_culture = polish
		}
		
		effect = {
			add_estate_loyalty = {
				estate = estate_polish_minorities
				loyalty = 10
			}
			
			every_owned_province = {
				limit = {
					culture = polish
				}
				add_local_autonomy = 25
				add_province_modifier = {
					name = ME_polish_nationalism
					duration = 7300
				}
			}
		}
	}
