declarewar = {
	condition = {
		tooltip = CANNOT_DECLARE_WAR_ON_OVERLORD_ALLY
		potential = {
			is_subject_of_type = protectorate
		}
		allow = {
			overlord = {
				NOT = { alliance_with = FROM }
				NOT = { has_guaranteed = FROM }
			}
		}
	}
	effect = {
		if = {
			limit = {
				FROM = {
					is_subject_of_type = protectorate
					overlord = {
						NOT = { war_with = ROOT }
					}
				}
			}
			FROM = {
				country_event = { id = ME_Protectorate_Events.3 days = 70 }
			}
		}
		if = {
			limit = {
				mission_completed = sweden_support_the_roman_church
				FROM = {
					NOT = { religion = catholic }
				}
			}
			change_variable = { swe_num_of_wars_against_non_catholics = 1 }
		}
	}
}
