HNS_kontor = {
	potential = {
		always = yes
	}

	trigger = {
		owner = {
			has_opinion = {
				who = event_target:target_HANSA
				value = 100
			}
		}
	}
	
	hidden = yes
	picture = "modifier_icon_lubeck_flag"
	
	#Province scope
	on_activation = {
		add_trade_modifier = {
			who = event_target:target_HANSA
			duration = -1
			power = 10
			key = HNS_ties
		}
		add_permanent_province_modifier = {
			name = HNS_icon
			duration = -1
			hidden = yes
		}
		remove_province_modifier = HNS_icon_gray
	}

	on_deactivation = {
		remove_province_modifier = HNS_icon
		add_permanent_province_modifier = {
			name = HNS_icon_gray
			duration = -1
			hidden = yes
		}
		remove_trade_modifier = {
			who = event_target:target_HANSA
			key = HNS_ties
		}
	}
}
