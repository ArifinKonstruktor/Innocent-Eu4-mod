prussia_make_prussian_vassal = {
	category = influence
	
	require_acceptance = no
	
	is_visible = {
		has_country_modifier = PRU_vassal_play
		FROM = {
			is_subject_of_type = vassal
			is_subject_of = ROOT
		}
	}
	
	is_allowed = {
		FROM = { is_at_war = no }
		if = {
			limit = {
				has_country_flag = PRU_vassal_more
			}
			NOT = { prussian_vassal = 5 }
		}
		else = {
			NOT = { prussian_vassal = 3 }
		}
	}
	
	on_accept = {
		create_subject = {
			subject_type = prussian_vassal 
			subject = FROM
		}
	}
	
	ai_will_do = {
	}
}
