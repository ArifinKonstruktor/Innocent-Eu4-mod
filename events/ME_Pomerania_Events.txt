country_event = {
	id = ME_Pomerania_Events.1 #Death of Stettiner Ruler
	title = ME_Pomerania_Events.1.t
	desc = ME_Pomerania_Events.1.d
	picture = COURT_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = STE
		ai = yes
		has_heir = no
		current_age = age_of_discovery
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = ME_Pomerania_Events.1.a
		
		ai_chance = { factor = 100 }
		kill_ruler = yes
		if = {
			limit = {
				exists = WOL
			}

				country_event = { id = ME_Pomerania_Events.2 }
			}
		}
		else_if = {
			limit = {
				exists = BRA
			}
			BRA = {
				country_event = { id = ME_Pomerania_Events.3 }
			}
		}
	}
}
