country_event = { # The Scouring of New Zealand
	id = ME_Australia_Events.3
	title = ME_Australia_Events.3.T
	desc = ME_Australia_Events.3.D
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		tag = AUS
		war_exhaustion = 0
		is_at_war = no
		te_ika_a_maui_area = {
			type = all
			is_core = ROOT
			NOT = {
				unrest = 3
			}
		}
		te_waipounamu_area = {
			type = all
			is_core = ROOT
			NOT = {
				unrest = 0
			}
		}
	}

	mean_time_to_happen = {
		months = 32
	}

	option = {
		name = ME_Australia_Events.3.OPT1

		ai_chance = {
			factor = 25
		}
		add_country_modifier = {
			name = aus_grazing_fields
			duration = -1
		}
		every_owned_province = {
			limit = {
				OR = {
					area = te_ika_a_maui_area
					area = te_waipounamu_area
				}
				OR = {
					has_terrain = grasslands
					has_terrain = farmlands
				}
			}
			add_devastation = 5
			random_list = {
				50 = {
					change_trade_goods = grain
				}
				50 = {
					change_trade_goods = livestock
				}
			}
		}
	}

	option = {
		name = ME_Australia_Events.3.OPT2

		ai_chance = {
			factor = 25
		}
		add_country_modifier = {
			name = aus_golden_woods
			duration = -1
		}
		every_owned_province = {
			limit = {
				OR = {
					area = te_ika_a_maui_area
					area = te_waipounamu_area
				}
				OR = {
					has_terrain = jungle
					has_terrain = forest
					has_terrain = woods
				}
			}
			add_devastation = 5
			random_list = {
				50 = {
					change_trade_goods = naval_supplies
				}
				50 = {
					change_trade_goods = tropical_wood
				}
			}
		}
	}

	option = {
		name = ME_Australia_Events.3.OPT3

		ai_chance = {
			factor = 25
		}
		add_country_modifier = {
			name = aus_eren_nimrais
			duration = -1
		}
		every_owned_province = {
			limit = {
				OR = {
					area = te_ika_a_maui_area
					area = te_waipounamu_area
				}
				OR = {
					has_terrain = hills
					has_terrain = mountain
					has_terrain = highlands
				}
			}
			add_devastation = 5
			random_list = {
				50 = {
					change_trade_goods = copper
				}
				50 = {
					change_trade_goods = iron
				}
			}
		}
	}

	option = {
		name = ME_Australia_Events.3.OPT4

		ai_chance = {
			factor = 25
		}
		every_owned_province = {
			limit = {
				OR = {
					area = te_ika_a_maui_area
					area = te_waipounamu_area
				}
			}
			add_devastation = -10
			add_unrest = +10
			add_province_modifier = {
				name = aus_scouring_stopped
				duration = -1
			}
		}
	}
}
