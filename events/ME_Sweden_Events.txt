country_event = {
	id = ME_Sweden_Events.1
	title = ME_Sweden_Events.1.t
	desc = ME_Sweden_Events.1.d
	picture = NEW_HEIR_eventPicture

	fire_only_once = yes

	trigger = {
		is_year = 1523
		tag = SWE
		government = monarchy
		is_lesser_in_union = no
		NOT = { has_reform = elective_monarchy }
		NOT = { has_reform = ottoman_government }
		NOT = { has_reform = mamluk_government }
		NOT = { has_reform = shogunate }
	}

	mean_time_to_happen = {
		months = 10
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = swe_gustav_vasa
		}
	}

	option = {
		name = ME_Sweden_Events.1.a

		ai_chance = { factor = 0.95 }

		define_ruler = {
			dynasty = Vasa
			age = 27
			fixed = yes
			adm = 4
			dip = 4
			mil = 4
			claim = 100
			female = no
			name = Gustav
		}
		define_ruler_to_general = {
			shock = 3
			fire = 3
			manuever = 3
			siege = 1
		}
	}

	option = {
		name = ME_Sweden_Events.1.b

		ai_chance = { factor = 0.05 }
	}
}
