#Ban of Croatia
country_event = {
	id = ME_Cilli_Events.1
	title = ME_Cilli_Events.1.T
	desc = ME_Cilli_Events.1.D
	picture = COURT_eventPicture
	
	is_triggered_only = yes

	option = {
		name = ME_Cilli_Events.1.OPT1
		
		ai_chance = { factor = 33 }

		HUN = {
			country_event = {
				id = ME_Cilli_Events.2
			}
		}

		custom_tooltip = Cilli_Austrian_Back
	}

	option = {
		name = ME_Cilli_Events.1.OPT2
		
		ai_chance = { factor = 33 }

		croatia_area = {
			limit = {
				NOT = { is_core = ROOT }
				NOT = { owned_by = ROOT }
			}
			add_permanent_claim = ROOT
		}
		slavonia_area = {
			limit = {
				NOT = { is_core = ROOT }
				NOT = { owned_by = ROOT }
			}
			add_permanent_claim = ROOT
		}
		east_adriatic_coast_area = {
			limit = {
				NOT = { is_core = ROOT }
				NOT = { owned_by = ROOT }
			}
			add_permanent_claim = ROOT
		}
	}

	option = {
		name = ME_Cilli_Events.1.OPT3
		trigger = {
			military_strength = {
				who = HUN
				value = 1
			}
		}

		set_country_flag = cli_intimidate_hungary

		HUN = {
			country_event = {
				id = ME_Cilli_Events.2
			}
		}
		
		ai_chance = { factor = 33 }
	}
}

#Fate of the Hunyadi
country_event = {
	id = ME_Cilli_Events.9
	title = ME_Cilli_Events.9.T
	desc = ME_Cilli_Events.9.D
	picture = COURT_eventPicture
	
	is_triggered_only = yes

	option = {
		name = ME_Cilli_Events.9.OPT1
		
		ai_chance = { factor = 50 }

		if = {
			limit = {
				has_dlc = "Rights of Man"
			}
			add_ruler_personality = cruel_personality
		}
		else = {
			add_stability_or_adm_power = yes
			add_ruler_modifier = {
				name = cli_executed_hunyadi
				duration = -1
			}
			HUN = {
				add_liberty_desire = -10
			}
		}
	}

	option = {
		name = ME_Cilli_Events.9.OPT2
		
		ai_chance = { factor = 50 }

		add_legitimacy_equivalent = {
			value = -10
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_opinion = {
			who = HUN
			modifier = cli_spared_hungarian_nobles
		}
	}
}
