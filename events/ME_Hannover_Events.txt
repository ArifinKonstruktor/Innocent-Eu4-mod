country_event = {
	id = ME_Hannover_Events.3
	title = ME_Hannover_Events.3.t
	desc = ME_Hannover_Events.3.d
	picture = BORDER_TENSION_eventPicture

	mean_time_to_happen = {
		months = 12
	}
	
	trigger = {
		tag = HAN
		owns = 4772
		OR = {
			had_country_flag = {
				flag = HAN_20_year_stud
				days = 7300
			}
			had_country_flag = {
				flag = HAN_40_year_stud
				days = 14600
			}
		}
	}
	
	option = {
		name = ME_Hannover_Events.3.a
		
		ai_chance = { factor = 1 }
		
		if = {
			limit = {
				owns = 4772
			}
			4772 = {
				add_province_modifier = {
					name = HAN_stud
					duration = -1
				}
				if = {
					limit = {
						is_expanded_mod_active = { mod = trade_goods }
					}
					change_trade_goods = horses
				}
			}
		}
		else = {
			random_owned_province = {
				add_province_modifier = {
					name = HAN_stud
					duration = -1
				}
				if = {
					limit = {
						is_expanded_mod_active = { mod = trade_goods }
					}
					change_trade_goods = horses
				}
			}
		}
	}
}
