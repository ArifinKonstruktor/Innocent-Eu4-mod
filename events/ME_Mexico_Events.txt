country_event = { #Native Contact
	id = ME_Mexico_Events.1
	title = ME_Mexico_Events.1.T
	desc = ME_Mexico_Events.1.D
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes

	trigger = {
		exists = AZT
		exists = TLC
	}

	option = {
		name = ME_Mexico_Events.1.OPT1

		ai_chance = {
			factor = 33
			modifier = {
				factor = 0
				alliance_with = TLC
				NOT = { alliance_with = AZT }
			}
		}

		if = {
			limit = {
				NOT = { alliance_with = AZT }
			}
			create_alliance = AZT
		}
		add_trust = {
			who = AZT
			value = 50
			mutual = yes
		}
		add_country_modifier = {
			name = mex_native_contact_modifier
			duration = 18250
		}
		TLC = {
			add_casus_belli = {
				type = cb_insult
				months = 240
				target = ROOT
			}
		}
	}

	option = {
		name = ME_Mexico_Events.1.OPT2

		ai_chance = {
			factor = 33
			modifier = {
				factor = 0
				alliance_with = AZT
				NOT = { alliance_with = TLC }
			}
		}

		if = {
			limit = {
				NOT = { alliance_with = TLC }
			}
			create_alliance = TLC
		}
		add_trust = {
			who = TLC
			value = 50
			mutual = yes
		}
		add_country_modifier = {
			name = mex_native_contact_modifier
			duration = 18250
		}
		AZT = {
			add_casus_belli = {
				type = cb_insult
				months = 240
				target = ROOT
			}
		}
	}

	option = {
		name = ME_Mexico_Events.1.OPT3

		ai_chance = {
			factor = 33
			modifier = {
				factor = 0
				alliance_with = AZT
				alliance_with = TLC
			}
		}

		TLC = {
			add_casus_belli = {
				type = cb_insult
				months = 240
				target = ROOT
			}
		}
		AZT = {
			add_casus_belli = {
				type = cb_insult
				months = 240
				target = ROOT
			}
		}
	}
}

country_event = { #Fate of Cholula
	id = ME_Mexico_Events.2
	title = ME_Mexico_Events.2.T
	desc = ME_Mexico_Events.2.D
	picture = CIVIL_WAR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = MEX
		owns = 2644
	}

	option = {
		name = ME_Mexico_Events.2.OPT1

		ai_chance = {
			factor = 33
		}

		add_years_of_income = 5
		2644 = {
			add_unrest = 10
			add_devastation = 20
		}
		every_country = {
			limit = {
				NOT = { tag = ROOT }
				capital_scope = {
					colonial_region = colonial_mexico
				}
				technology_group = mesoamerican
			}

				who = ROOT
				modifier = mex_razed_sacred_city
			}
		}
