country_event = { #Jianzhou - Ginseng Production
	id = ME_Manchu_Events.1
	title = ME_Manchu_Events.1.t
	desc = ME_Manchu_Events.1.d
	picture = COURT_eventPicture

	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		random_owned_province = {
			limit = {
				base_production = 10
			}
			save_event_target_as = ginseng_province
		}
	}
	
	option = {
		name = ME_Manchu_Events.1.a
		
		event_target:ginseng_province = {
			change_trade_goods = tea
			add_province_modifier = {
				name = MCH_ginseng
				duration = -1
			}
		}
	}
	
	option = {
		name = ME_Manchu_Events.1.b
		
		trigger = {
			event_target:ginseng_province = {
				is_capital = no
			}
		}
		
		capital_scope = {
			change_trade_goods = tea
			add_province_modifier = {
				name = MCH_ginseng
				duration = -1
			}
		}
	}
}
