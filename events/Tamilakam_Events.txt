#The Tamil Sangams
country_event = {
    id = ME_Tamilakam_Events.4
    title = ME_Tamilakam_Events.4.a
    desc = ME_Tamilakam_Events.4.b
    picture = 2_Brahmin_Estate_eventPicture
       
	trigger = {
		mission_completed = tamil_reestablish_sangams
		had_country_flag = { flag = tamil_sangam_event days = 7300 }
		any_owned_province = {
			culture = tamil
			is_capital = no
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			clr_country_flag = tamil_sangam_event
			set_country_flag = tamil_sangam_event
		}
	}
	
	#Support the Sangam this time
    option = {
        name = ME_Tamilakam_Events.4.c
        random_owned_province = {
			limit = { culture = tamil is_capital = no }
			move_capital_effect = yes
		}
		add_years_of_income = -1.5
		add_yearly_manpower = -1.0
		add_country_modifier = {
            name = "tamil_new_sangam_gathering"
            duration = 7300 #20 Years
        } 
    }
	#Does not support the Sangam this time
	option = {
        name = ME_Tamilakam_Events.4.d
		add_years_of_income = 0.5
		add_adm_power = 50
		add_country_modifier = {
            name = "tamil_denied_sangam_gathering"
            duration = 7300 #20 Years
        }
    }
}
