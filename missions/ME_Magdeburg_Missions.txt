			MAG_find_alliances
		}
		position = 2

		}

				}
			}	
		}
	}
	MAG_saxon_frontier = {
		icon = mission_imperial
		required_missions = { MAG_conquest_of_anhalt }
		position = 3
		provinces_to_highlight = {
			owned_by = SAX	
		}	
		trigger = {
			every_province = {
				if = {

		}
		effect = {
			if = {
				limit = { num_of_electors = 6 }
				elector = ROOT
			}
			else = {
				add_claim = 50
			}	
		}
	}
