Odoyev_slot_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = OKA
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	OKA_capital_fort = {
		icon = mission_conquer_50_development
		position = 1
		required_missions = { }
		trigger = {
			capital_scope = {
				fort_level = 3
			}	
		}
		effect = {
			capital_scope = {
				add_province_modifier = {
					name = OKA_defended_capital
					duration = 7300
				}
			}	
		}	
	}
	OKA_monastery_of_holy_mother = {
		icon = mission_european_church
		position = 2
		required_missions = { }
		trigger = {
			4116 = {
				OR = {
					has_building = temple
					has_building = cathedral
				}
			}	
		}
		effect = {
			if = { 
				limit = { 
					4116 = {
						has_building = cathedral
					}	
				}
				add_permanent_province_modifier = {
					name = OKA_building_religious_stuffs
					duration = -1
				}	
			}
			if = { 
				limit = { 
					4116 = {
						has_building = temple
						NOT = { has_building = cathedral }
						can_build = cathedral
					}	
				}
				OKA = { country_event = { id = odoyev_event.1 } }
			}	
			4116 = {
				add_permanent_province_modifier = {
					name = OKA_building_religious_stuffs
					duration = 7300
				}
			}	
		}
	}
}

					}
				}

					NOT = { is_rival = OKA }

					}

			}
