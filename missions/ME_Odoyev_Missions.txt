OKA_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = OKA
		OR = {
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
			NOT = { has_dlc = "Third Rome" }
		}
	}
	has_country_shield = yes

	OKA_capital_fort = {
		icon = mission_conquer_50_development
		position = 1
		required_missions = { }
		provinces_to_highlight = {
			is_capital_of = ROOT
			NOT = { fort_level = 3 }
		}
		trigger = {
			capital_scope = {
				fort_level = 3
			}
		}
		effect = {
			capital_scope = {
				add_province_modifier = {
					name = OKA_defended_capital
					duration = 7300
				}
			}
		}
	}

	OKA_monastery_of_holy_mother = {
		icon = mission_european_church
		position = 2
		required_missions = { }
		provinces_to_highlight = {
			province_id = 4116
			OR = {
				NOT = { owned_by = ROOT }
				AND = {
					NOT = { has_building = temple }
					NOT = { has_building = cathedral }
				}
			}
		}
		trigger = {
			4116 = {
				owned_by = ROOT
				OR = {
					has_building = temple
					has_building = cathedral
				}
			}
		}
		effect = {
			4116 = {
				if = {
					limit = {
						has_building = cathedral
					}
					add_permanent_province_modifier = {
						name = OKA_building_religious_stuffs
						duration = -1
					}
				}
				else_if = {
					limit = {
						has_building = temple
						NOT = { has_building = cathedral }
						can_build = cathedral
					}
					country_event = { id = ME_Odoyev_Events.1 }
						add_permanent_province_modifier = {
						name = OKA_building_religious_stuffs
						duration = 7300
					}
				}
				else = {
					add_permanent_province_modifier = {
						name = OKA_building_religious_stuffs
						duration = 3650
					}
				}
			}
		}
	}
}

		position = 1
		required_missions = { }
		trigger = {
			OR = {
				exists = RUS
				exists = MOS
			}
			if = {
				limit = {
					exists = RUS
				}
				NOT = { is_rival = RUS }
				NOT = { is_enemy = RUS }
				reverse_has_opinion = {
					who = RUS
					value = 100
				}
			}
			else = {
				NOT = { is_rival = MOS }
				NOT = { is_enemy = MOS }
				reverse_has_opinion = {
					who = MOS
					value = 100
				}
			}
			NOT = { mission_completed = OKA_align_crimea }
			NOT = { mission_completed = OKA_align_mishar }
			NOT = { mission_completed = OKA_align_lithuania }
		}
		effect = {
			if = {
				limit = {
					exists = RUS
				}
				add_trust = {
					who = RUS
					value = 5
					mutual = yes
				}
			}
			else = {
				add_trust = {
					who = MOS
					value = 5
					mutual = yes
				}
			}
		}
	}

	OKA_ally_russia = {
		icon = me_mission_muscovy_flag
		position = 2
		required_missions = { OKA_align_russia }
		trigger = {
			if = {
				limit = {
					exists = RUS
				}
				alliance_with = RUS
			}
			else = {
				alliance_with = MOS
			}
		}
		effect = {
			if = {
				limit = {
					exists = RUS
				}
				RUS = {
					create_guarantee = ROOT
				}
			}
			else = {
				MOS = {
					create_guarantee = ROOT
				}
			}
			add_country_modifier = {
				name = OKA_dip_rep
				duration = 3650
			}
		}
	}
}
