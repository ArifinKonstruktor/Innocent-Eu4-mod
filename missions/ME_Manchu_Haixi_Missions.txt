haixi_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			was_tag = MHX
			tag = MHX
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MHX_ming = {
		icon = mission_manchu_warrior
		required_missions = {}
		position = 1
		trigger = {
			OR = {
				NOT = { exists = MNG }
				is_subject_of = MNG
				AND = {
					is_neighbor_of = MNG
					reverse_has_opinion = {
						who = MNG
						value = 50
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = MCH_ming_trade
				duration = 9125
			}
		}
	}
	
	MHX_nomadic_traditions = {
		icon = mission_manchu_warrior
		required_missions = {
			MHX_jurchen_eat_mjz
		}
		position = 3
		provinces_to_highlight = {
			province_id = 4201
			NOT = { is_core = ROOT }
		}
		trigger = {
			stability = 1
			4201 = {
				is_core = ROOT
			}
			OR = {
				employed_advisor = { type = army_reformer }
				employed_advisor = { type = army_organiser }
				employed_advisor = { type = commandant }
				employed_advisor = { type = quartermaster }
				employed_advisor = { type = recruitmaster }
				employed_advisor = { type = fortification_expert }
				employed_advisor = { type = grand_captain }
			}
		}
		effect = {
			add_country_modifier = {
				name = MCH_mhx_nomads
				duration = 9125
			}
		}
	}
}

haixi_missions_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = MHX
			tag = MCH
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MHX_jurchen_eat_mjz = {
		icon = mission_manchu_warrior
		required_missions = {
			MHX_jurchen_great_leader
		}
		position = 2
		provinces_to_highlight = {
			owned_by = MJZ
		}
		trigger = {
			NOT = { exists = MJZ }
		}
		effect = {
			if = {
				limit = {
					NOT = { mission_completed = MHX_jurchen_eat_myr }
				}
				add_country_modifier = {
					name = MCH_unification_ambitions
					duration = 3650
				}
			}
			if = {
				limit = {
					mission_completed = MHX_jurchen_eat_myr
					has_dlc = "Mandate of Heaven"
				}
				add_country_modifier = {
					name = MCH_chinese_raids
					duration = 9125
				}
				
			}
			else_if = {
				limit = {
					mission_completed = MHX_jurchen_eat_myr
					NOT = { has_dlc = "Mandate of Heaven" }
				}
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
			else = {
				manchuria_region = {
					limit = {
						NOT = { owned_by = ROOT }
						NOT = { is_core = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
}

	}
	has_country_shield = yes
	
	MHX_jurchen_great_leader = {
		icon = mission_manchu_warrior
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				num_of_banners = 5
			}
			army_size_percentage = 1.0
			OR = {
				is_monarch_leader = yes
				is_heir_leader = yes
				has_leader_with = { shock = 4 }
			}
		}
		effect = {
			add_corruption = -2
			every_owned_province = {
				every_neighbor_province = {
					limit = {
						region = manchuria_region
						NOT = { owned_by = ROOT }
						NOT = { is_core = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	
	MCH_jurchen_region = {
		icon = mission_manchu_warrior
		required_missions = {
			MHX_jurchen_great_leader
		}
		provinces_to_highlight = {
			region = manchuria_region
			NOT = { owned_by = ROOT }
		}
		position = 3
		trigger = { #copied from vanilla
			tag = MCH
			custom_trigger_tooltip = {
				tooltip = manchu_manchuria.tooltip
				num_of_owned_provinces_with = {
					value = 25
					region = manchuria_region
				}
			}
		}
		effect = {
			add_ruler_modifier = { #copied from vanilla
				name = "uniter_of_the_manchus"
				duration = -1
			}
		}
	}
}
