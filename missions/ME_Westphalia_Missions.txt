WES_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = WES
	}
	has_country_shield = yes
	
	wes_tolerance = {
		icon = mission_religious
		trigger = {
			dip_tech = 12
			prestige = 30
		}
		effect = {
			add_country_modifier = {
				name = wes_tolerant_society
				duration = -1
			}
		}
	}
	
	wes_jews = {
		icon = mission_religious
		required_missions = { wes_tolerance }
		position = 2
		trigger = {
			dip_tech = 14
			dip_power = 150
		}
		effect = {
			add_dip_power = -150
			add_country_modifier = {
				name = wes_jewish_enclave
				duration = -1
			}
		}
	}
	
	wes_parliament = {
		icon = mission_monarch_in_throne_room
		required_missions = { wes_jews }
		position = 3
		trigger = {
			legitimacy_equivalent = 100
			adm_power = 100
			prestige = 50
		}
		effect = {
			if = {
				limit = {
					has_dlc = "Dharma"
				}
				if = {
					limit = {
						NOT = { has_reform = parliamentary_reform }
					}
					add_government_reform = parliamentary_reform
				}
				else = {
					add_legitimacy_equivalent = { value = 10 }
					every_owned_province = {
						add_local_autonomy = -10
					}
				}
			}
			else = {
				if = {
					limit = {
						government = monarchy
					}
					set_legacy_government = constitutional_monarchy
				}
				else_if = {
					limit = {
						government = republic
					}
					set_legacy_government = constitutional_republic
				}
				else = {
					add_legitimacy_equivalent = { value = 10 }
					every_owned_province = {
						add_local_autonomy = -10
					}
				}
			}
		}
	}
	
	wes_constitution = {
		icon = mission_alliances
		required_missions = { wes_parliament }
		position = 4
		trigger = {
			adm_tech = 20
			adm_power = 300
		}
		effect = {
			add_country_modifier = {
				name = wes_westphalian_constitution
				duration = -1
			}
		}
	}
	
	wes_industrial = {
		icon = mission_have_manufactories
		required_missions = { wes_constitution }
		position = 5
		trigger = {
			if = {
				limit = {
					has_dlc = "Rule Britannia"
				}
				furnace = 5
			}
			else = {
				num_of_owned_manufactory_trigger = { amount = 5 }
			}
			adm_tech = 25	
		}	
		effect = {
			add_prestige = 50
			add_country_modifier = {
				name = wes_industrial_revolution
				duration = -1
			}
		}
	}
}
