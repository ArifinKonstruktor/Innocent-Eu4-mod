ASK_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = ASK
		OR = {
			AND = {
				NOT = { has_reform = indep_daimyo }
				NOT = { has_reform = daimyo }
				NOT = { has_reform = shogunate }
			}
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
		}
	}
	has_country_shield = yes
	
	ASK_the_kyoto_contribution = {
		icon = mission_high_income
		position = 1
		required_missions = { }
		trigger = {
			custom_trigger_tooltip = {
				tooltip = ME_Ashikaga_Tooltips.3
				has_country_flag = JAP_daimyo_contribution_flag
			}
		}
		effect = {
			add_stability = 1
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				remove_ruler_personality = obsessive_perfectionist_personality
				add_ruler_personality = architectural_visionary_personality
			}
			else = {
				capital_scope = {				
					add_province_modifier = {
						name = ASK_daimyo_tributes
						duration = 7300
					}
				}
			}
		}
	}
	ASK_development_plan = {
		icon = mission_asian_city
		position = 3
		required_missions = { ASK_the_kyoto_contribution ASK_enforce_diplomatic_attendance }
		provinces_to_highlight = {
			is_capital_of = ROOT
			OR = {
				AND = {
					NOT = { has_building = temple }
					NOT = { has_building = cathedral }
				}
				AND = { 
					NOT = { has_building = marketplace }
					NOT = { has_building = trade_depot }
					NOT = { has_building = stock_exchange }
				}
			}
		}
		trigger = {
			capital_scope = {
				OR = {
					has_building = temple
					has_building = cathedral
				}
				OR = {
					has_building = marketplace
					has_building = trade_depot
					has_building = stock_exchange
				}
			}
		}
		effect = {
			capital_scope = {
				add_institution_embracement = {
					which = renaissance
					value = 30
				}
				
			}
			if = {
				limit = {
					OR = {
						NOT = { has_dlc = "Mandate of Heaven" }
						capital_scope = {
							is_prosperous = yes
						}
					}
				}
				add_years_of_income = 0.33
			}
			else = {
				capital_scope = {
					add_prosperity = 100
				}
			}
		}
	}
	ASK_temple = {
		icon = mission_rb_trinity_college_dublin
		position = 4
		required_missions = { ASK_development_plan }
		provinces_to_highlight = {
			is_capital_of = ROOT
			NOT = { development = 35 }
		}
		trigger = {
			capital_scope = {
				development = 35
			}
		}
		effect = {
			capital_scope = {
				add_province_modifier = {
					name = ASK_golden_pavilion
					duration = -1
				}
			}
		}
	}
}

	ASK_restore_shogun_authority = {
		icon = mission_kowtow
		position = 1
		required_missions = { }
		trigger = {
			OR = {
				diplomat = 1
				statesman = 1
			}
			NOT = {
				any_country = {
					is_subject_of = ROOT
					liberty_desire = 30
				}
			}
		}
		effect = {
			add_legitimacy = 5
			add_country_modifier = {
				name = ASK_Reinstated
				duration = 7300
			}
		}
	}
	ASK_enforce_diplomatic_attendance  = {
		icon = mission_scholar_officials
		position = 2
		required_missions = { ASK_restore_shogun_authority }
		trigger = {
			all_subject_country = {
				has_opinion = {
					who = ROOT
					value = 125
				}
			}
			if = {
				limit = {
					is_expanded_mod_active = { mod = governments }
					has_dlc = "Dharma"
				}
				OR = {
					has_reform = diplomatic_positions_reform
					diplomat = 2
				}
			}
			else = {
				diplomat = 2
			}	
		}
		effect = {
			change_dip = 1
			add_prestige = 10
			add_country_modifier = {
				name = ASK_enforced_attendance
				duration = 9125
			}
		}
	}
	ASK_diplomatic_integration  = {
		icon = mission_rb_monopolize_the_channel
		position = 3
		required_missions = { ASK_enforce_diplomatic_attendance }
