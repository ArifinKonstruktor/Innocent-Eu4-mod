jianzhou_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			was_tag = MJZ
			tag = MJZ
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MJZ_ming = {
		icon = mission_manchu_warrior
		required_missions = {}
		position = 1
		trigger = {
			OR = {
				NOT = { exists = MNG }
				is_subject_of = MNG
				AND = {
					any_neighbor_country = {
						tag = MNG
					}
					reverse_has_opinion = {
						who = MNG
						value = 50
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = MCH_ming_trade
				duration = 9125
			}
		}
	}
	
	MJZ_ginseng = {
		icon = mission_manchu_warrior
		required_missions = {
			MJZ_ming
		}
		position = 2
		trigger = {
			stability = 1
			any_owned_province = {
				base_production = 10
			}
		}
		effect = {
			country_event = {
				id = ME_Manchu_Events.1
			}
		}
	}
}

jianzhou_missions_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = MJZ
			tag = MCH
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MJZ_jurchen_eat_mhx = {
		icon = mission_manchu_warrior
		required_missions = {
			MJZ_jurchen_great_leader
		}
		provinces_to_highlight = {
			owned_by = MHX
		}
		position = 2
		trigger = {
			NOT = {
				exists = MHX
			}
		}
		effect = {
			IF = {
				NOT = {
					mission_completed = MJZ_jurchen_eat_myr
				}
				add_country_modifier = {
					name = MCH_unification_ambitions
					duration = 3650
				}
			}
			IF = {
				limit = {
					mission_completed = MJZ_jurchen_eat_myr
					has_dlc = "Mandate of Heaven"
				}
				add_country_modifier = {
					name = MCH_chinese_raids
					duration = 9125
				}
				
			}
			ELSE_IF = {
				limit = {
					mission_completed = MJZ_jurchen_eat_myr
					NOT = { has_dlc = "Mandate of Heaven" }
				}
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
			ELSE = {
				manchuria_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
}

jianzhou_missions_3 = {
	slot = 3
	generic = no

	potential = {

			tag = MJZ
			tag = MCH
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MJZ_jurchen_great_leader = {
		icon = mission_manchu_warrior
		required_missions = { }
		position = 1
		trigger = {
			IF = {
				limit = { has_dlc = "Mandate of Heaven" }
				num_of_banners = 5
			}
			army_size_percentage = 1.0
			OR = {
				is_monarch_leader = yes
				is_heir_leader = yes
				has_leader_with = { shock = 4 }
			}
		}
		effect = {
			add_corruption = -2
			every_owned_province = {
				every_neighbor_province = {
					limit = {
						region = manchuria_region
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	
	MJZ_jurchen_region = {
		icon = mission_manchu_warrior
		required_missions = {
			MJZ_jurchen_great_leader
		}
		provinces_to_highlight = {
			region = manchuria_region
			NOT = { owned_by = ROOT }
		}
		position = 3
		trigger = { #copied from vanilla
			tag = MCH
			custom_trigger_tooltip = {
				tooltip = manchu_manchuria.tooltip
				num_of_owned_provinces_with = {
					region = manchuria_region
					value = 25
				}
			}
		}
		effect = {
			add_ruler_modifier = { #copied from vanilla
				name = "uniter_of_the_manchus"
			}
		}
	}
}
