	generic = no

	potential = {
		tag = YMN
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes

	Yaman_Ascendacy = { # Ascendacy
		icon = mission_asian_trader
		required_missions = { }
		position = 1
		trigger = {
			is_at_war = yes
			is_monarch_leader = yes
			OR = {
				1020 = {
					owner = {
						reverse_has_opinion = {
							who = ROOT
							value = 125
						}
					}
				}
				1020 = {
					owned_by = ROOT
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = YMN_Daimyo_Politics
				duration = 7300
			}
			sanindo_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			saigoku_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	Yamana_Silk_for_Blood = { # Silk for Blood
		icon = mission_impaled_soldiers
		required_missions = { Yaman_Ascendacy }
		position = 2
		provinces_to_highlight = {
			area = saigoku_area
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			saigoku_area = {
				type = all
				owned_by = ROOT
			}
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 3
				area = sanindo_area
			}
		}
		effect = {
			1017 = {
				add_base_tax = 1
				add_base_manpower = 1
				add_base_production = 1
			}
			kyushu_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			northern_kyushu = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	Yamana_Lord_of_Kyushu = { # Lord of Kyushu
		icon = mission_warchest
		required_missions = { Yamana_Silk_for_Blood }
		position = 3
		provinces_to_highlight = {
			OR = {
				area = kyushu_area
				area = northern_kyushu
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 4
				OR = {
					area = kyushu_area
					area = northern_kyushu
				}
				NOT = {
					province_id = 1015
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = YMN_Kyushu_Trade_Ports
				duration = 7300
			}
			add_prestige = 5
		}
	}
	Yamana_Secure_our_Trade_Route = { # Secure our Trade Routes
		icon = mission_seize_the_dragon_throne
		required_missions = { 
			Yamana_Lord_of_Kyushu
			Yamana_Shugo_of_South_Japan
		}
		position = 5
		trigger = {
			OR = {
				AND = {
					4651 = {
						owned_by = ROOT
					}
					navy_size_percentage = 1
					light_ship_fraction = 0.25
				}
				AND = {
					navy_size_percentage = 1
					calc_true_if = {
						all_owned_province = {
							province_has_center_of_trade_of_level = 2
						}
						amount = 2
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					4651 = {
						owned_by = ROOT
					}
					navy_size_percentage = 1
					light_ship_fraction = 0.25
					is_expanded_mod_active = TGE
				}
				add_dip_power = 50
				add_country_modifier = {
					name = YMN_Safe_Merchant_Haven
					duration = 9125
				}
				4651 = {
					change_trade_goods = pearls
				}
			}
			else_if = {
				limit = {
					4651 = {
						owned_by = ROOT
					}
					navy_size_percentage = 1
					light_ship_fraction = 0.25
				}
				add_dip_power = 50
				add_country_modifier = {
					name = YMN_Safe_Merchant_Haven
					duration = 9125
				}
			}
			else_if = {
				limit = {
					navy_size_percentage = 1
					calc_true_if = {
						all_owned_province = {
							province_has_center_of_trade_of_level = 2
						}
						amount = 2
					}
				}
				add_country_modifier = {
					name = YMN_Pirate_Patrols
					duration = 7300
				}
				add_dip_power = 100
			}
		}
	}
}
