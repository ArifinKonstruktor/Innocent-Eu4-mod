YMN_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = YMN
		OR = {
			AND = {
				NOT = { has_reform = indep_daimyo }
				NOT = { has_reform = daimyo }
				NOT = { has_reform = shogunate }
			}
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
		}
	}
	has_country_shield = yes

	Yaman_Ascendacy = { # Ascendacy
		icon = mission_asian_trader
		required_missions = { }
		position = 1
		provinces_to_highlight = {
			province_id = 1020
			NOT = { owned_by = ROOT }
			owner = {
				NOT = {
					has_opinion = {
						who = ROOT
						value = 125
					}
				}
			}
		}
		trigger = {
			is_at_war = yes
			is_monarch_leader = yes
			1020 = {
				OR = {
					owned_by = ROOT
					owner = {
						has_opinion = {
							who = ROOT
							value = 125
						}
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = YMN_Daimyo_Politics
				duration = 7300
			}
			sanindo_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			saigoku_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	Yamana_Silk_for_Blood = { # Silk for Blood
		icon = mission_impaled_soldiers
		required_missions = { Yaman_Ascendacy }
		position = 2
		provinces_to_highlight = {
			OR = {
				area = saigoku_area
				AND = {
					area = sanindo_area
					ROOT = {
						NOT = {
							num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
								value = 3
								area = sanindo_area
							}
						}
					}
				}
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		trigger = {
			saigoku_area = {
				type = all
				owned_by = ROOT
			}
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 3
				area = sanindo_area
			}
		}
		effect = {
			1017 = {
				add_base_tax = 1
				add_base_manpower = 1
				add_base_production = 1
			}
			kyushu_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			northern_kyushu = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	Yamana_Lord_of_Kyushu = { # Lord of Kyushu
		icon = mission_warchest
		required_missions = { Yamana_Silk_for_Blood }
		position = 3
		provinces_to_highlight = {
			OR = {
				area = kyushu_area
				area = northern_kyushu
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
			ROOT = {
				NOT = {
					num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
						value = 4
						OR = {
							area = kyushu_area
							area = northern_kyushu
						}
					}
				}
			}
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 4
				OR = {
					area = kyushu_area
					area = northern_kyushu
				}
			}
		}
		effect = {
			add_prestige = 5
			add_country_modifier = {
				name = YMN_Kyushu_Trade_Ports
				duration = 7300
			}
		}
	}
	
	Yamana_Secure_our_Trade_Route = { # Secure our Trade Routes
		icon = mission_seize_the_dragon_throne
		required_missions = { 
			Yamana_Lord_of_Kyushu
			Yamana_Shugo_of_South_Japan
		}
		position = 5
		provinces_to_highlight = {
			province_id = 4651
			NOT = { owned_by = ROOT }
			ROOT = {
				NOT = {
					num_of_owned_provinces_with = {
						value = 2
						province_has_center_of_trade_of_level = 2
					}
				}
			}
		}
		trigger = {
			navy_size_percentage = 1
			OR = {
				num_of_owned_provinces_with = {
					value = 2
					province_has_center_of_trade_of_level = 2
				}
				AND = {
					light_ship_fraction = 0.25
					owns = 4651
				}
			}
		}
		effect = {
			if = {
				limit = {
					owns = 4651
				}
				add_dip_power = 50
				add_country_modifier = {
					name = YMN_Safe_Merchant_Haven
					duration = 9125
				}
				if = {
					limit = {
						is_expanded_mod_active = { mod = trade_goods }
					}
					4651 = {
						change_trade_goods = pearls
					}
				}
			}
			else = {
				add_dip_power = 100
				add_country_modifier = {
					name = YMN_Pirate_Patrols
					duration = 7300
				}
			}
		}
	}
}
