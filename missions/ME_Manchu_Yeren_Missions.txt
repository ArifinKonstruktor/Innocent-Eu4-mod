yeren_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			was_tag = MYR
			tag = MYR
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MYR_unite_yeren = {
		icon = mission_manchu_warrior
		required_missions = {}
		provinces_to_highlight = {
			owned_by = ROOT
			has_terrain = mountains
			NOT = { units_in_province = ROOT }
		}
		position = 1
		trigger = {
			stability = 1
			legitimacy_equivalent = 90
			all_owned_province = {
				OR = {
					NOT = { has_terrain = mountains }
					units_in_province = ROOT
				}
			}
		}
		effect = {
			east_heilongjiang_area = {
				limit = {
					owned_by = ROOT
				}
				add_base_manpower = 1
			}
			ilan_hala_area = {
				limit = {
					owned_by = ROOT
				}
				add_base_manpower = 1
			}
			central_jilin_area = {
				limit = {
					owned_by = ROOT
				}
				add_base_manpower = 1
			}
		}
	}
	
	MYR_buriats = {
		icon = mission_manchu_warrior
		required_missions = {
			MYR_unite_yeren
		}
		position = 2
		trigger = {
			OR = {
				NOT = { exists = BRT }
				overlord_of = BRT
			}
		}
		effect = {
			east_siberia_region = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			east_siberia_region = {
				discover_country = ROOT
			}
		}
	}
	
	MYR_siberian_tribes = {
		icon = mission_manchu_warrior
		required_missions = {
			MYR_buriats
		}
		position = 3
		provinces_to_highlight = {
			region = east_siberia_region
			is_empty = no
			NOT = { owned_by = ROOT }
		}
		trigger = {
			east_siberia_region = {
				type = all
				OR = {
					owned_by = ROOT
					is_empty = yes
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = MCH_siberian_traditions
				duration = -1
			}
		}
	}
}

yeren_missions_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = MYR
			tag = MCH
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MYR_jurchen_eat_mjz = {
		icon = mission_manchu_warrior
		required_missions = {
			MYR_jurchen_great_leader
		}
		position = 2
		trigger = {
			NOT = { exists = MJZ }
		}
		effect = {
			if = {
				limit = {
					NOT = { mission_completed = MYR_jurchen_eat_mhx }
				}
				add_country_modifier = {
					name = MCH_unification_ambitions
					duration = 3650
				}
			}
			if = {
				limit = {
					mission_completed = MYR_jurchen_eat_mhx
					has_dlc = "Mandate of Heaven"
				}
				add_country_modifier = {
					name = MCH_chinese_raids
					duration = 9125
				}
				
			}
			else_if = {
				limit = {
					mission_completed = MYR_jurchen_eat_mhx
					NOT = { has_dlc = "Mandate of Heaven" }
				}
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
			else = {
				manchuria_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
}

yeren_missions_3 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		OR = {

			tag = MCH
		}
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
	}
	has_country_shield = yes
	
	MYR_jurchen_great_leader = {
		icon = mission_manchu_warrior
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				num_of_banners = 5
			}
			army_size_percentage = 1.0
			OR = {
				is_monarch_leader = yes
				is_heir_leader = yes
				has_leader_with = { shock = 4 }
			}
		}
		effect = {
			add_corruption = -2
			every_owned_province = {
				every_neighbor_province = {
					limit = {
						region = manchuria_region
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	
	MCH_jurchen_region = {
		icon = mission_manchu_warrior
		required_missions = {
			MYR_jurchen_great_leader
		}
		provinces_to_highlight = {
			region = manchuria_region
			NOT = { owned_by = ROOT }
		}
		position = 3
		trigger = { #copied from vanilla
			tag = MCH
			custom_trigger_tooltip = {
				tooltip = manchu_manchuria.tooltip
				num_of_owned_provinces_with = {
					region = manchuria_region
					value = 25
				}
			}
		}
		effect = {
			add_ruler_modifier = { #copied from vanilla
				name = "uniter_of_the_manchus"
				duration = -1
			}
		}
	}
}
